> [!info]- 需安装 **Dataview** 插件并开启 *JavaScript Queries*（见 [[说明]]）。分类自动发现；误收的日期文件夹加进 `EXCLUDE`。

```dataviewjs
const EXCLUDE = ["看板", "dashboards", "templates", ".trash"];
const PALETTE = ["#1971c2","#e8590c","#2f9e44","#9c36b5","#c2255c","#f08c00","#0c8599","#5f3dc4","#e03131","#1098ad","#d6336c","#66a80f"];
const esc = s => String(s).replace(/[&<>"]/g, m => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;" }[m]));
const rgba = (hex, a) => { const n = parseInt(hex.slice(1), 16); return `rgba(${n>>16&255},${n>>8&255},${n&255},${a})`; };
const short = c => c.length > 4 ? c.slice(0, 4) + "…" : c;

function collect() {
  const out = [];
  for (const p of dv.pages()) {
    if (!/^\d{4}-\d{2}-\d{2}$/.test(p.file.name)) continue;
    const category = p.file.folder.split("/").pop();
    if (!category || EXCLUDE.includes(category)) continue;
    const d = dv.luxon.DateTime.fromISO(p.file.name);
    if (!d.isValid) continue;
    const tf = app.vault.getAbstractFileByPath(p.file.path);
    const cache = tf ? app.metadataCache.getFileCache(tf) : null;
    const heads = (cache && cache.headings) ? cache.headings.filter(h => h.level === 2) : [];
    for (const h of heads) out.push({ date: d, category });
  }
  return out;
}

const all = collect();
const cats = [...new Set(all.map(e => e.category))].sort();
const catColor = {}; cats.forEach((c, i) => catColor[c] = PALETTE[i % PALETTE.length]);
const col = c => catColor[c] || "#868e96";
const now = dv.luxon.DateTime.now();
const days = new Set(all.map(e => e.date.toFormat("yyyy-MM-dd"))).size;

const tot = cats.map(c => [c, all.filter(x => x.category === c).length]).sort((a, b) => b[1] - a[1]);
const maxTot = tot.length ? Math.max(1, tot[0][1]) : 1;
const rankBars = tot.map(([c, n]) => `
  <div class="jo-bar-row"><span class="jo-cat"><i class="jo-dot" style="background:${col(c)}"></i>${esc(c)}</span>
  <span class="jo-track"><i class="jo-fill" style="width:${Math.max(4, n / maxTot * 100)}%;background:${col(c)}"></i></span>
  <span class="jo-cnt">${n}</span></div>`).join("");

function heatmap(buckets) {
  let gmax = 1;
  const grid = buckets.map(b => cats.map(c => {
    const n = all.filter(x => x.category === c && x.date >= b.start && x.date <= b.end).length;
    gmax = Math.max(gmax, n); return n;
  }));
  const head = `<tr><th>区间</th><th>合计</th>${cats.map(c => `<th title="${esc(c)}">${esc(short(c))}</th>`).join("")}</tr>`;
  const body = buckets.map((b, i) => {
    const row = grid[i]; const sum = row.reduce((a, x) => a + x, 0);
    const cells = row.map((n, j) => {
      const c = cats[j];
      const bg = n ? rgba(col(c), 0.18 + 0.82 * (n / gmax)) : "transparent";
      const fg = n && (n / gmax) > 0.55 ? "#fff" : "var(--text-normal)";
      return `<td style="background:${bg};color:${fg}">${n || ""}</td>`;
    }).join("");
    return `<tr><td class="jo-rl">${esc(b.label)}</td><td class="jo-sum">${sum || ""}</td>${cells}</tr>`;
  }).join("");
  return `<table class="jo-hm"><thead>${head}</thead><tbody>${body}</tbody></table>`;
}

const weeks = [];
for (let i = 7; i >= 0; i--) { const s = now.minus({ weeks: i }).startOf("week"); weeks.push({ label: s.toFormat("M/d"), start: s, end: s.endOf("week") }); }
const months = [];
for (let i = 5; i >= 0; i--) { const s = now.minus({ months: i }).startOf("month"); months.push({ label: s.toFormat("yyyy/M"), start: s, end: s.endOf("month") }); }

const wrap = dv.el("div", "");
wrap.innerHTML = `
<style>
.jo-board{font-size:.92em}
.jo-board .jo-head{display:flex;align-items:baseline;gap:10px;margin:2px 0 6px}
.jo-board .jo-h1{font-size:1.5em;font-weight:700}
.jo-board .jo-sub{color:var(--text-muted)}
.jo-board .jo-st{font-weight:600;color:var(--text-muted);font-size:.85em;letter-spacing:.05em;text-transform:uppercase;margin:18px 0 8px}
.jo-board .jo-bar-row{display:flex;align-items:center;gap:10px;margin:6px 0}
.jo-board .jo-cat{width:120px;flex-shrink:0;display:flex;align-items:center;gap:6px;font-size:.92em}
.jo-board .jo-dot{width:9px;height:9px;border-radius:50%;display:inline-block;flex-shrink:0}
.jo-board .jo-track{flex:1;height:9px;background:var(--background-modifier-border);border-radius:6px;overflow:hidden}
.jo-board .jo-fill{display:block;height:100%;border-radius:6px}
.jo-board .jo-cnt{width:26px;text-align:right;color:var(--text-muted);font-variant-numeric:tabular-nums}
.jo-board table.jo-hm{border-collapse:separate;border-spacing:3px;width:100%;font-variant-numeric:tabular-nums}
.jo-board .jo-hm th{font-weight:600;color:var(--text-muted);font-size:.82em;text-align:center;padding:2px 0}
.jo-board .jo-hm td{text-align:center;border-radius:6px;padding:6px 4px;font-size:.85em}
.jo-board .jo-hm .jo-rl{text-align:left;color:var(--text-muted);white-space:nowrap}
.jo-board .jo-hm .jo-sum{font-weight:700;background:var(--background-secondary)}
</style>
<div class="jo-board">
  <div class="jo-head"><span class="jo-h1">📊 总览</span><span class="jo-sub">累计 ${all.length} 条 · 跨 ${days} 天</span></div>
  <div class="jo-st">累计分类排行</div>${rankBars}
  <div class="jo-st">近 8 周（颜色越深越高产）</div>${heatmap(weeks)}
  <div class="jo-st">近 6 个月</div>${heatmap(months)}
</div>`;
```
