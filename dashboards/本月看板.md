> [!info]- 需安装 **Dataview** 插件并开启 *JavaScript Queries*（见 [[说明]]）。分类自动发现；误收的日期文件夹加进 `EXCLUDE`。

```dataviewjs
const EXCLUDE = ["看板", "dashboards", "templates", ".trash"];
const PALETTE = ["#1971c2","#e8590c","#2f9e44","#9c36b5","#c2255c","#f08c00","#0c8599","#5f3dc4","#e03131","#1098ad","#d6336c","#66a80f"];
const esc = s => String(s).replace(/[&<>"]/g, m => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;" }[m]));

function collect() {
  const out = [];
  for (const p of dv.pages()) {
    if (!/^\d{4}-\d{2}-\d{2}$/.test(p.file.name)) continue;
    const category = p.file.folder.split("/").pop();
    if (!category || EXCLUDE.includes(category)) continue;
    const d = dv.luxon.DateTime.fromISO(p.file.name);
    if (!d.isValid) continue;
    const tf = app.vault.getAbstractFileByPath(p.file.path);
    const cache = tf ? app.metadataCache.getFileCache(tf) : null;
    const heads = (cache && cache.headings) ? cache.headings.filter(h => h.level === 2) : [];
    for (const h of heads) {
      const m = h.heading.match(/^(\d{1,2}:\d{2})\s*[·:：]\s*(.*)$/);
      out.push({ date: d, category, time: m ? m[1] : "", title: m ? m[2] : h.heading, path: p.file.path, heading: h.heading });
    }
  }
  return out;
}

const all = collect();
const cats = [...new Set(all.map(e => e.category))].sort();
const catColor = {}; cats.forEach((c, i) => catColor[c] = PALETTE[i % PALETTE.length]);
const col = c => catColor[c] || "#868e96";

const now = dv.luxon.DateTime.now();
const ms = now.startOf("month"), me = now.endOf("month");
const mo = all.filter(e => e.date >= ms && e.date <= me)
  .sort((a, b) => a.date.toMillis() - b.date.toMillis() || a.time.localeCompare(b.time));

const byCat = {}; for (const e of mo) byCat[e.category] = (byCat[e.category] || 0) + 1;
const ranked = Object.entries(byCat).sort((a, b) => b[1] - a[1]);
const maxc = ranked.length ? ranked[0][1] : 1;
const activeDays = new Set(mo.map(e => e.date.toFormat("yyyy-MM-dd"))).size;

const byWeek = {};
for (const e of mo) { const k = e.date.startOf("week").toFormat("M/d"); byWeek[k] = (byWeek[k] || 0) + 1; }
const weekRows = Object.entries(byWeek);
const maxw = weekRows.length ? Math.max(...weekRows.map(r => r[1])) : 1;

const cards = `
  <div class="jo-card"><div class="jo-num">${mo.length}</div><div class="jo-lbl">条记录</div></div>
  <div class="jo-card"><div class="jo-num">${activeDays}</div><div class="jo-lbl">活跃天</div></div>
  <div class="jo-card"><div class="jo-num" style="font-size:1.05em">${ranked.length ? esc(ranked[0][0]) : "—"}</div><div class="jo-lbl">最高产</div></div>`;

const bars = ranked.map(([c, n]) => `
  <div class="jo-bar-row"><span class="jo-cat"><i class="jo-dot" style="background:${col(c)}"></i>${esc(c)}</span>
  <span class="jo-track"><i class="jo-fill" style="width:${Math.max(8, n / maxc * 100)}%;background:${col(c)}"></i></span>
  <span class="jo-cnt">${n}</span></div>`).join("");

const wbars = weekRows.map(([k, n]) => `
  <div class="jo-bar-row"><span class="jo-cat" style="width:90px">${k} 这周</span>
  <span class="jo-track"><i class="jo-fill" style="width:${Math.max(8, n / maxw * 100)}%;background:var(--text-accent)"></i></span>
  <span class="jo-cnt">${n}</span></div>`).join("");

const groups = ranked.map(([c]) => {
  const items = mo.filter(e => e.category === c).map(e => `
    <a class="internal-link jo-entry" data-href="${esc(e.path.replace(/\.md$/, "") + "#" + e.heading)}">
      <span class="jo-time">${e.date.toFormat("M/d")} ${e.time}</span><span class="jo-et">${esc(e.title)}</span></a>`).join("");
  return `<div class="jo-group"><div class="jo-gh"><i class="jo-dot" style="background:${col(c)}"></i>${esc(c)}<span class="jo-gc">${byCat[c]}</span></div>${items}</div>`;
}).join("");

const wrap = dv.el("div", "");
wrap.innerHTML = `
<style>
.jo-board{font-size:.92em}
.jo-board .jo-head{display:flex;align-items:baseline;gap:10px;margin:2px 0 14px}
.jo-board .jo-h1{font-size:1.5em;font-weight:700}
.jo-board .jo-sub{color:var(--text-muted)}
.jo-board .jo-cards{display:flex;gap:10px;margin-bottom:18px;flex-wrap:wrap}
.jo-board .jo-card{flex:1;min-width:90px;background:var(--background-secondary);border:1px solid var(--background-modifier-border);border-radius:12px;padding:12px 14px;text-align:center}
.jo-board .jo-num{font-size:1.7em;font-weight:700;line-height:1.1}
.jo-board .jo-lbl{color:var(--text-muted);font-size:.85em;margin-top:2px}
.jo-board .jo-st{font-weight:600;color:var(--text-muted);font-size:.85em;letter-spacing:.05em;text-transform:uppercase;margin:18px 0 8px}
.jo-board .jo-bar-row{display:flex;align-items:center;gap:10px;margin:6px 0}
.jo-board .jo-cat{width:120px;flex-shrink:0;display:flex;align-items:center;gap:6px;font-size:.92em}
.jo-board .jo-dot{width:9px;height:9px;border-radius:50%;display:inline-block;flex-shrink:0}
.jo-board .jo-track{flex:1;height:9px;background:var(--background-modifier-border);border-radius:6px;overflow:hidden}
.jo-board .jo-fill{display:block;height:100%;border-radius:6px}
.jo-board .jo-cnt{width:26px;text-align:right;color:var(--text-muted);font-variant-numeric:tabular-nums}
.jo-board .jo-group{margin:4px 0 14px}
.jo-board .jo-gh{display:flex;align-items:center;gap:7px;font-weight:600;margin:10px 0 6px}
.jo-board .jo-gc{margin-left:6px;background:var(--background-modifier-border);border-radius:10px;padding:0 8px;font-size:.78em;color:var(--text-muted);font-weight:500}
.jo-board .jo-entry{display:flex;align-items:baseline;gap:10px;padding:7px 11px;border-radius:9px;text-decoration:none!important;color:var(--text-normal)!important;border:1px solid transparent}
.jo-board .jo-entry:hover{background:var(--background-secondary);border-color:var(--background-modifier-border)}
.jo-board .jo-time{flex-shrink:0;color:var(--text-muted);font-size:.82em;font-variant-numeric:tabular-nums;min-width:64px}
.jo-board .jo-et{font-size:.95em}
.jo-board .jo-empty{color:var(--text-muted);padding:8px 0}
</style>
<div class="jo-board">
  <div class="jo-head"><span class="jo-h1">🗓️ 本月</span><span class="jo-sub">${ms.toFormat("yyyy 年 M 月")}</span></div>
  <div class="jo-cards">${cards}</div>
  ${mo.length ? `<div class="jo-st">分类分布</div><div class="jo-bars">${bars}</div>` : ""}
  ${mo.length ? `<div class="jo-st">按周分布</div><div class="jo-bars">${wbars}</div>` : ""}
  <div class="jo-st">条目</div>
  ${mo.length ? groups : `<div class="jo-empty">本月还没有记录 ✍️</div>`}
</div>`;

wrap.querySelectorAll(".jo-entry").forEach(a => a.addEventListener("click", ev => {
  ev.preventDefault();
  app.workspace.openLinkText(a.dataset.href, dv.current().file.path, ev.ctrlKey || ev.metaKey);
}));
```
